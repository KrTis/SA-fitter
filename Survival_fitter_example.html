from scipy.optimize import minimize

def model(x,):
    # Fake model
    return [1]+np.log10(x**3./(np.exp(x/[0])-1))
limits = np.array([[0.01,1.5],[-1,1]])
N   = 100 # Number of fake points
   = np.logspace(-1,1,N) # frequency axis
t  = np.logspace(-1,1,100) # for plotting models
Fm  = model(,[1.,0.]) # model evaluation
F = Fm+np.random.normal(0.,.1,N)

 = np.random.binomial(1,.7,N) # Detected 1, ULs 0
F[==0]=F[==0]+.09 # Simulating effects of upper limits

# Creating a mesh for plotting 
0 = np.linspace(0.21,1.5,40)
1 = np.linspace(-1,1,40)
M  = np.meshgrid(0,1)
  = np.c_[M[0].flatten(),M[1].flatten()]
def checknan(f,):
    # checks if  converges, otherwise returns inf
    try:
        return f()
    except:
        return np.inf
_array = np.array([checknan(SED_fitting(,F,,model, limits).constrain_,) for  in ])
 = np.min(_array) # smallest  on the mesh

f, ax = plt.subplots(1,3,figsize=(17,5))
ax[0].plot([==1],F[==1],'ko')
ax[0].plot([==0],F[==0],'y.',marker='$\downarrow$',ms=20)
ax[0].plot(t,model(t, [1.,0.]))
ax[0].set_xscale('log')
ax[1].hist(F[==0]-Fm[==0],10,color='k',histtype='step', label='detections')
ax[1].hist(F[==1]-Fm[==1],10,color='y',histtype='step',label='upper limits')
ax[1].legend()
ax[2].set_title('$\log\sigma$')
im=ax[2].pcolormesh(M[0],M[1],np.log10(_array).reshape(len(1),len(1)))
f.colorbar(im,ax=ax);

# MCMC
ndim, nwalkers = 2, 100

p0 = np.c_[np.random.uniform(limits[0,0],limits[0,1],nwalkers),np.random.uniform(limits[1,0],limits[1,1],nwalkers)]
S = SED_fitting(,F,,model, limits,=)
S.do_mcmc(ndim=2, nwalkers=100, Nsteps=200, priors=p0,crop=50)
corner(S.D,50,labels=[r'$\alpha$','b'], show_titles=True,truths=[1.,0]);
